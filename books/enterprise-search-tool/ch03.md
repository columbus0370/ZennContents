## 3-2. APIクライアント

`src/api/client.js`:
```js
const API_BASE = import.meta.env.VITE_API_URL || "http://localhost:8000";

async function apiFetch(path, options = {}) {
  const res = await fetch(`${API_BASE}${path}`, {
    headers: { "Content-Type": "application/json" },
    ...options,
  });
  if (!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

export const fetchIndustries = () => apiFetch("/api/industries");

export const searchCompanies = (params) =>
  apiFetch("/api/search", {
    method: "POST",
    body: JSON.stringify(params),
  });

export const generateProposal = (companyInfo) =>
  apiFetch("/api/generate-proposal", {
    method: "POST",
    body: JSON.stringify(companyInfo),
  });
```

## 3-3. useHistoryフック（localStorage永続化）

`src/hooks/useHistory.js`:
```js
import { useState } from "react";

const SEARCH_KEY = "enterprise_search_history";
const PROPOSAL_KEY = "generated_proposals";
const MAX_ITEMS = 10;

function loadFromStorage(key) {
  try {
    return JSON.parse(localStorage.getItem(key) || "[]");
  } catch {
    return [];
  }
}

function saveToStorage(key, items) {
  localStorage.setItem(key, JSON.stringify(items.slice(0, MAX_ITEMS)));
}

export function useHistory() {
  const [searchHistory, setSearchHistory] = useState(() =>
    loadFromStorage(SEARCH_KEY)
  );
  const [proposalHistory, setProposalHistory] = useState(() =>
    loadFromStorage(PROPOSAL_KEY)
  );

  const addSearch = (query, resultsCount) => {
    const entry = {
      id: `search_${Date.now()}`,
      timestamp: Date.now(),
      query,
      results_count: resultsCount,
    };
    const next = [entry, ...searchHistory];
    setSearchHistory(next);
    saveToStorage(SEARCH_KEY, next);
  };

  const addProposal = (companyName, proposalText, subject) => {
    const entry = {
      id: `proposal_${Date.now()}`,
      company_name: companyName,
      proposal_text: proposalText,
      subject,
      generated_at: Date.now(),
    };
    const next = [entry, ...proposalHistory];
    setProposalHistory(next);
    saveToStorage(PROPOSAL_KEY, next);
  };

  const clearAll = () => {
    setSearchHistory([]);
    setProposalHistory([]);
    localStorage.removeItem(SEARCH_KEY);
    localStorage.removeItem(PROPOSAL_KEY);
  };

  return { searchHistory, proposalHistory, addSearch, addProposal, clearAll };
}
```

## 3-4. SearchFormコンポーネント

`src/components/SearchForm.jsx`:
```jsx
import { useState, useEffect } from "react";
import { fetchIndustries } from "../api/client";

const PREFECTURES = [
  "東京都", "大阪府", "神奈川県", "愛知県", "埼玉県",
  "千葉県", "兵庫県", "福岡県", "北海道", "静岡県",
  "茨城県", "広島県", "京都府", "宮城県", "新潟県",
];

export default function SearchForm({ onSearch, isLoading }) {
  const [industries, setIndustries] = useState([]);
  const [form, setForm] = useState({
    industry: "",
    employee_min: 0,
    employee_max: 1000,
    prefecture: "東京都",
  });
  const [error, setError] = useState("");

  useEffect(() => {
    fetchIndustries()
      .then((res) => setIndustries(res.data))
      .catch(() => setIndustries([]));
  }, []);

  const handleSubmit = () => {
    if (!form.industry) {
      setError("業種を選択してください");
      return;
    }
    if (form.employee_min > form.employee_max) {
      setError("従業員数の下限が上限を超えています");
      return;
    }
    setError("");
    onSearch(form);
  };

  return (
    <div className="bg-navy-900 p-6 rounded-lg space-y-4">
      <h2 className="text-white text-lg font-medium">企業検索</h2>

      {error && <p className="text-red-400 text-sm">{error}</p>}

      <div>
        <label className="text-gray-300 text-sm block mb-1">業種 *</label>
        <select
          className="w-full bg-navy-700 text-white rounded px-3 py-2"
          value={form.industry}
          onChange={(e) => setForm({ ...form, industry: e.target.value })}
        >
          <option value="">選択してください</option>
          {industries.map((ind) => (
            <option key={ind} value={ind}>{ind}</option>
          ))}
        </select>
      </div>

      <div>
        <label className="text-gray-300 text-sm block mb-1">都道府県</label>
        <select
          className="w-full bg-navy-700 text-white rounded px-3 py-2"
          value={form.prefecture}
          onChange={(e) => setForm({ ...form, prefecture: e.target.value })}
        >
          {PREFECTURES.map((pref) => (
            <option key={pref} value={pref}>{pref}</option>
          ))}
        </select>
      </div>

      <div className="grid grid-cols-2 gap-3">
        <div>
          <label className="text-gray-300 text-sm block mb-1">従業員数（下限）</label>
          <input
            type="number"
            className="w-full bg-navy-700 text-white rounded px-3 py-2"
            value={form.employee_min}
            min={0}
            onChange={(e) =>
              setForm({ ...form, employee_min: Number(e.target.value) })
            }
          />
        </div>
        <div>
          <label className="text-gray-300 text-sm block mb-1">従業員数（上限）</label>
          <input
            type="number"
            className="w-full bg-navy-700 text-white rounded px-3 py-2"
            value={form.employee_max}
            min={0}
            onChange={(e) =>
              setForm({ ...form, employee_max: Number(e.target.value) })
            }
          />
        </div>
      </div>

      <button
        className="w-full bg-accent-500 hover:bg-blue-400 text-white py-2 rounded font-medium transition-colors disabled:opacity-50"
        onClick={handleSubmit}
        disabled={isLoading}
      >
        {isLoading ? "検索中..." : "企業を検索する"}
      </button>
    </div>
  );
}
```

## 3-5. ProposalEditorコンポーネント

`src/components/ProposalEditor.jsx`:
```jsx
import { useState } from "react";

export default function ProposalEditor({ subject, proposalText, onRegenerate }) {
  const [isEditing, setIsEditing] = useState(false);
  const [editedText, setEditedText] = useState(proposalText);
  const [copied, setCopied] = useState(false);

  const handleCopy = async () => {
    const fullText = `件名: ${subject}\n\n${editedText}`;
    await navigator.clipboard.writeText(fullText);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  return (
    <div className="bg-navy-900 rounded-lg p-6 space-y-4">
      <div className="flex items-center justify-between">
        <h3 className="text-white font-medium">生成されたメール</h3>
        <div className="flex gap-2">
          <button
            className="text-sm bg-navy-700 hover:bg-navy-500 text-white px-3 py-1 rounded transition-colors"
            onClick={() => setIsEditing(!isEditing)}
          >
            {isEditing ? "完了" : "編集"}
          </button>
          <button
            className="text-sm bg-accent-500 hover:bg-blue-400 text-white px-3 py-1 rounded transition-colors"
            onClick={handleCopy}
          >
            {copied ? "コピーしました！" : "コピー"}
          </button>
          <button
            className="text-sm bg-navy-700 hover:bg-navy-500 text-white px-3 py-1 rounded transition-colors"
            onClick={onRegenerate}
          >
            再生成
          </button>
        </div>
      </div>

      <div className="bg-navy-700 rounded p-3">
        <span className="text-gray-400 text-sm">件名: </span>
        <span className="text-white text-sm">{subject}</span>
      </div>

      {isEditing ? (
        <textarea
          className="w-full bg-navy-700 text-white rounded p-3 text-sm leading-relaxed resize-none"
          rows={15}
          value={editedText}
          onChange={(e) => setEditedText(e.target.value)}
        />
      ) : (
        <div className="bg-navy-700 rounded p-3 text-white text-sm leading-relaxed whitespace-pre-wrap">
          {editedText}
        </div>
      )}
    </div>
  );
}
```

## 3-6. App.jsx（全体統合）

`src/App.jsx`:
```jsx
import { useState } from "react";
import SearchForm from "./components/SearchForm";
import ProposalEditor from "./components/ProposalEditor";
import { searchCompanies, generateProposal } from "./api/client";
import { useHistory } from "./hooks/useHistory";

export default function App() {
  const [results, setResults] = useState([]);
  const [selectedCompany, setSelectedCompany] = useState(null);
  const [proposal, setProposal] = useState(null);
  const [isSearching, setIsSearching] = useState(false);
  const [isGenerating, setIsGenerating] = useState(false);
  const [activeTab, setActiveTab] = useState("search");
  const { searchHistory, proposalHistory, addSearch, addProposal } = useHistory();

  const handleSearch = async (params) => {
    setIsSearching(true);
    setResults([]);
    try {
      const res = await searchCompanies(params);
      if (res.status === "success") {
        setResults(res.data);
        addSearch(params, res.total_count);
        setActiveTab("results");
      }
    } catch {
      alert("検索に失敗しました");
    } finally {
      setIsSearching(false);
    }
  };

  const handleGenerateProposal = async (company) => {
    setSelectedCompany(company);
    setIsGenerating(true);
    setProposal(null);
    try {
      const res = await generateProposal({
        company_name: company.name,
        industry: company.industry,
        employees: company.employees_from,
        prefecture: company.prefecture,
      });
      if (res.status === "success") {
        setProposal(res.data);
        addProposal(company.name, res.data.proposal_text, res.data.subject);
      }
    } catch {
      alert("メール生成に失敗しました");
    } finally {
      setIsGenerating(false);
    }
  };

  return (
    <div className="min-h-screen bg-gray-950 text-white">
      <header className="bg-navy-900 border-b border-navy-700 px-6 py-4">
        <h1 className="text-xl font-medium text-accent-500">AI営業アシスタント</h1>
      </header>

      <div className="flex border-b border-navy-700 bg-navy-900">
        {["search", "results", "history"].map((tab) => (
          <button
            key={tab}
            className={`px-6 py-3 text-sm transition-colors ${
              activeTab === tab
                ? "text-accent-500 border-b-2 border-accent-500"
                : "text-gray-400 hover:text-white"
            }`}
            onClick={() => setActiveTab(tab)}
          >
            {{ search: "検索", results: "結果", history: "履歴" }[tab]}
          </button>
        ))}
      </div>

      <main className="max-w-6xl mx-auto p-6">
        {activeTab === "search" && (
          <SearchForm onSearch={handleSearch} isLoading={isSearching} />
        )}

        {activeTab === "results" && (
          <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
            <div>
              <p className="text-gray-400 text-sm mb-3">
                {results.length}件の企業が見つかりました
              </p>
              <div className="space-y-2">
                {results.map((company) => (
                  <div
                    key={company.id}
                    className="bg-navy-900 rounded-lg p-4 hover:bg-navy-700 transition-colors"
                  >
                    <div className="flex items-start justify-between">
                      <div>
                        <p className="text-white font-medium">{company.name}</p>
                        <p className="text-gray-400 text-sm">
                          {company.industry} | {company.prefecture}
                        </p>
                        {company.employees_from > 0 && (
                          <p className="text-gray-400 text-sm">
                            従業員 {company.employees_from}〜{company.employees_to}名
                          </p>
                        )}
                      </div>
                      <button
                        className="bg-accent-500 hover:bg-blue-400 text-white text-sm px-3 py-1 rounded transition-colors whitespace-nowrap ml-3 disabled:opacity-50"
                        onClick={() => handleGenerateProposal(company)}
                        disabled={isGenerating}
                      >
                        メール生成
                      </button>
                    </div>
                  </div>
                ))}
              </div>
            </div>

            <div>
              {isGenerating && (
                <div className="bg-navy-900 rounded-lg p-6 text-center text-gray-400">
                  AIがメールを生成中...
                </div>
              )}
              {proposal && !isGenerating && (
                <ProposalEditor
                  subject={proposal.subject}
                  proposalText={proposal.proposal_text}
                  onRegenerate={() =>
                    selectedCompany && handleGenerateProposal(selectedCompany)
                  }
                />
              )}
            </div>
          </div>
        )}

        {activeTab === "history" && (
          <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
            <div>
              <h3 className="text-white font-medium mb-3">検索履歴</h3>
              {searchHistory.length === 0 ? (
                <p className="text-gray-500 text-sm">履歴はありません</p>
              ) : (
                searchHistory.map((h) => (
                  <div key={h.id} className="bg-navy-900 rounded p-3 mb-2 text-sm">
                    <p className="text-white">
                      {h.query.industry} / {h.query.prefecture}
                    </p>
                    <p className="text-gray-400">
                      {new Date(h.timestamp).toLocaleDateString("ja-JP")} —{" "}
                      {h.results_count}件
                    </p>
                  </div>
                ))
              )}
            </div>
            <div>
              <h3 className="text-white font-medium mb-3">生成メール履歴</h3>
              {proposalHistory.length === 0 ? (
                <p className="text-gray-500 text-sm">履歴はありません</p>
              ) : (
                proposalHistory.map((h) => (
                  <div key={h.id} className="bg-navy-900 rounded p-3 mb-2 text-sm">
                    <p className="text-white">{h.company_name}</p>
                    <p className="text-gray-400 truncate">{h.subject}</p>
                    <p className="text-gray-500">
                      {new Date(h.generated_at).toLocaleDateString("ja-JP")}
                    </p>
                  </div>
                ))
              )}
            </div>
          </div>
        )}
      </main>
    </div>
  );
}
```
